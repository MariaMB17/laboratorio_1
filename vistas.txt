SELECT examenes.idexamen as idexamen,examenes.decripcion as decripcion, CASE v_referencia_ex when '' then examen_subgrupo.v_referencia_sg else examenes.v_referencia_ex END as valor_referencia, CASE examen_subgrupo.`descripcion_sg` when null then 'No tiene' else examen_subgrupo.`descripcion_sg` END as valor_referencia1, examen_subgrupo.`descripcion_sg` as descripcion_sg, examenes.precio as precio FROM examen_subgrupo right JOIN examenes on examen_subgrupo.idexamen= examenes.idexamen
